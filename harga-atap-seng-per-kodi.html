        t = 9m

                                = [tex] \sqrt{144+81} [/tex]

L selimut = 22/7 x 12cm x 15cm
              = 565,7[tex] m^{2} [/tex]

Biaya pembelian seng = Rp 14.000,00/m2 x 565,2 m2
                                 = Rp 7.919.800,00 (kemungkinan dibulatkan) = Rp 7.920.000,00
