MAPEL = MATEMATIKA
KELAS = 9
BAB     = BARISAN & DERET

diketahui :
U7   = 37
U11 = 57
n     = 20

ditanya = banyak kursi pada baris ke-20 ( U20 )??

-------- Barisan Aritmatika -------

U7 = 37
a + 6b = 37.........( 1)

U11 = 57
a + 10b = 57.......( 2)

eliminasi nilai a pada pers. ( 1) dan ( 2)
  a + 6b = 37
a + 10b = 57
------------------  -
       -4b = -20
          b = -20/-4
          b = 5

subtitusi nilai b pada pers. ( 1)
   a + 6b = 37
a + 6(5) = 37
           a = 37 - 30
           a = 7

U20 = a + ( n - 1 )b
U20 = 7 + ( 20 - 1)(5)
U20 = 7 + 95
U20 = 102

jadi banyak kursi pada baris ke-20 = 102
